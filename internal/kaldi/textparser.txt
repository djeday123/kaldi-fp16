package kaldi

import (
	"bufio"
	"fmt"
	"os/exec"
	"strconv"
	"strings"
)

// TextExample - пример из текстового вывода Kaldi
type TextExample struct {
	Key         string
	NumInputs   int
	InputName   string
	InputFrames int
	InputDim    int
	FirstRow    []float32
}

// GetTextOutput запускает nnet3-chain-copy-egs и возвращает текстовый вывод
func GetTextOutput(arkFile string, maxExamples int) ([]TextExample, error) {
	cmd := exec.Command("bash", "-c",
		fmt.Sprintf("nnet3-chain-copy-egs 'ark:%s' ark,t:- 2>/dev/null | head -n 50000", arkFile))
	output, err := cmd.Output()
	if err != nil {
		return nil, fmt.Errorf("nnet3-chain-copy-egs failed: %v", err)
	}
	return ParseTextOutput(string(output), maxExamples)
}

// ParseTextOutput парсит текстовый вывод Kaldi
func ParseTextOutput(text string, maxExamples int) ([]TextExample, error) {
	var examples []TextExample
	var current *TextExample
	matrixRow := 0

	scanner := bufio.NewScanner(strings.NewReader(text))
	buf := make([]byte, 0, 1024*1024)
	scanner.Buffer(buf, 10*1024*1024)

	for scanner.Scan() {
		line := scanner.Text()

		// Новый пример: "key <Nnet3ChainEg> <NumInputs> 2"
		if strings.Contains(line, "<Nnet3ChainEg>") {
			if current != nil {
				examples = append(examples, *current)
				if len(examples) >= maxExamples {
					return examples, nil
				}
			}
			current = &TextExample{}
			matrixRow = 0

			parts := strings.Fields(line)
			if len(parts) > 0 {
				current.Key = parts[0]
			}
			for i, p := range parts {
				if p == "<NumInputs>" && i+1 < len(parts) {
					current.NumInputs, _ = strconv.Atoi(parts[i+1])
				}
			}
			continue
		}

		// Input header: "<NnetIo> input <I1V> 224 ..."
		if strings.Contains(line, "<NnetIo>") && current != nil && current.InputName == "" {
			parts := strings.Fields(line)
			for i, p := range parts {
				if p == "<NnetIo>" && i+1 < len(parts) {
					current.InputName = parts[i+1]
				}
				if p == "<I1V>" && i+1 < len(parts) {
					current.InputFrames, _ = strconv.Atoi(parts[i+1])
				}
			}
			continue
		}

		// Строка матрицы
		trimmed := strings.TrimSpace(line)
		if current != nil && current.InputName != "" && len(trimmed) > 0 {
			if strings.HasPrefix(trimmed, "[") || (len(trimmed) > 0 && (trimmed[0] == '-' || (trimmed[0] >= '0' && trimmed[0] <= '9'))) {
				trimmed = strings.TrimPrefix(trimmed, "[")
				trimmed = strings.TrimSuffix(trimmed, "]")
				trimmed = strings.TrimSpace(trimmed)

				if matrixRow == 0 && len(current.FirstRow) == 0 && len(trimmed) > 0 {
					parts := strings.Fields(trimmed)
					current.InputDim = len(parts)
					for _, p := range parts {
						if v, err := strconv.ParseFloat(p, 32); err == nil {
							current.FirstRow = append(current.FirstRow, float32(v))
						}
					}
				}
				matrixRow++
			}
		}
	}

	if current != nil && len(examples) < maxExamples {
		examples = append(examples, *current)
	}
	return examples, nil
}
